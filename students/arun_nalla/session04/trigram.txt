def trigram_dict (words):
    trilist = {}
    for i in range(len(words)-2):
        a = tuple(words[i:i+2])
        b = words[i+2]
        if a not in trilist:
            trilist[a] = [b]
        else:
            trilist[a].append(b)
    return trilist
